cmake_minimum_required(VERSION 3.10) # Specify the minimum required CMake version

project(MyCppProject VERSION 1.0) # Define your project name and version

# Set the C++ standard to use (e.g., C++17, C++20)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add an executable target
add_executable(MyCppProject main.cpp) # Create an executable named MyCppProject from main.cpp
